  1. Casting is when a variable has a type in parenthesis beside it. The compiler will attempt to coerce that variable to the given type in the parenthesis (e.g. (int)myVar). If the cast is valid, the type coercion will succeed. Otherwise either the compiler will throw an error, or the program will throw an error at runtime.

    What will be the type of result? 

    object foo = "Test"; // object can store any type

Given this method for the following questions:

public static class Homework
{
    public static bool HitSucceeds(int proficiencyBonus, int modifier, int enemyArmorClass)
    {
        var rand = new Random();
        var diceRoll = rand.Next(1, 20);
        
        if (diceRoll == 1)
        {
            // Critical miss
            return false;
        }
        if (diceRoll == 20)
        {
            // Critical hit
            return true;
        }
        // Apply modifiers to the dice roll
        diceRoll += proficiencyBonus + modifier;
        return diceRoll > enemyArmorClass;
    }
}

  2. Which player has the best chance to defeat their monster first? 

    public class Monster
{
    public int Health { get; set; }
    public int ArmorClass { get; set; }
}
public class Player
{
    public int Modifier { get; set; }
    
    public int ProficencyBonus { get; set; }
}
var monsters = new Monster[4];
for (var x = 0; x < 4; x++)
{
    monsters[x] = new Monster { Health = 100; ArmorClass = 10 };
}
var players = new Player[]
              {
                new Player { Modifier = 2, ProficiencyBonus = 2 },
                new Player { Modifier = 0, ProficiencyBonus = 4 },
                new Player { Modifier = 4, ProficiencyBonus = 1 },
                new Player { Modifier = 3, ProficiencyBonus = 1 }
              };
var winner = -1;
while (winner == -1)
{
    for (var x = 0; x < 4; x++)
    {
        // Current player
        var p = players[x];
        // Current monster
        var m = monsters[x];
        if (Homework.HitSucceeds(p.ProficiencyBonus, p.Modifier, m.ArmorClass)) 
        {
            // Hit!
            m.Health -= 10;
            
            if (m.Health <= 0)
            {
                winner = x;
                break;
            }
        }
    }
}
Console.WriteLine("Player " + winner + " defeated their monster first!");
